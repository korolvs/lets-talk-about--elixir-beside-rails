Code example

plug :find_user

def show(conn, %{"id" => id}) do
 post = conn.assigns.user |> assoc(:posts) |> Repo.get(id)
 render conn, "show.html", post: post
end

defp find_user(conn, _) do
 assign(conn, :user, Repo.get(User, conn.params["user_id"]))
end
